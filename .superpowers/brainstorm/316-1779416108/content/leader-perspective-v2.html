更新后的数据模型 — 新增 DataSourceInfo

models.py 中新增数据源元信息模型
class DataSourceInfo(BaseModel): # 数据源元信息（领导视角用） id: str name: str # 数据源名称，如"BENTI_DB" type: str # "dameng" | "excel" | "csv" conn_info: dict # {host, port, schema} 或 {file_name} tables: list[str] # 包含的表/Sheet/文件名 covered_nodes: int # 映射的本体概念数 covered_domains: list[str] # 支撑的业务域ID列表 total_fields: int # 总字段数 class LeaderViewData(BaseModel): # 领导视角完整数据 summary: dict # {domains, nodes, edges, sources} data_sources: list[DataSourceInfo] # 数据源清单 domains: list[BusinessDomain] # 业务域列表 source_domain_map: dict # 数据源→业务域映射
